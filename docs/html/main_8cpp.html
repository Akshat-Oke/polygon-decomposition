vector< Vertex * > decompose ( vector< Vertex * >  vertices)

Decompose a polygon into convex polygons.

Parameters
verticesVertices in clockwise order
Returns
vector<Vertex *>

v(1) = Last[L]

v(2) = Next[P, v(1)]

L = {v(1), v(2)}

i = 2

v(i+1) = Next[P, v(i)]

L = L + {v(i+1)}

i = i + 1

v(i+1) = Next[P, v(i)]

P = P - L

v = First(LPVS)

LPVS = LPVS - {v}
